26static esp_err_t lcd_send_nibble(uint8_t nibble, bool rs) {
27 uint8_t data_base = LCD_PIN_BACKLIGHT | ((nibble & 0x0F) << 4);
28 if (rs) data_base |= LCD_PIN_RS;
29
30 uint8_t seq[3];
31 seq[0] = data_base;
32 seq[1] = data_base | LCD_PIN_EN;
33 seq[2] = data_base;
34
35 esp_err_t err = i2c_master_transmit(g_dev_handle, seq, 3, 100);
36 vTaskDelay(pdMS_TO_TICKS(2));
37 return err;
38}

40static void lcd_send_byte(uint8_t value, bool rs) {

63void lcd_init(void) {

66 .i2c_port = I2C_NUM_0,
67 .scl_io_num = LCD_SCL_GPIO,

69 .glitch_ignore_cnt = 7,
70 .flags.enable_internal_pullup = true,

82 // Inicialización manual (sin mutex aún porque no hay tareas corriendo)
