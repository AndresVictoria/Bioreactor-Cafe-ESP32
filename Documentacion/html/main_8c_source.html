101void tarea_control_proceso(void *pvParameters) {
102 datos_fermentacion_t datos_vivos;
103 int ticks_en_alerta = 0;
104 int toggle_visual = 0;
105 rgb_init();
106 botones_init();
107
108 while (1) {
109 boton_mezcla_presionado();
110
111 if (xSemaforoMezcla != NULL && xSemaphoreTake(xSemaforoMezcla, 0) == pdTRUE) {
112 seg_desde_ultima_mezcla = 0;
113 mezcla_manual_hecha = true;
114
115 // CORRECCIÓN: Forzamos la limpieza de la cola para que la pantalla se actualice YA
116 if (xQueuePeek(xColaDatos, &datos_vivos, 0) == pdTRUE) {
117 if (datos_vivos.causa_alerta == 6) {
118 datos_vivos.causa_alerta = 0;
119 xQueueOverwrite(xColaDatos, &datos_vivos);
120 }
121 }
122 printf("[EVENTO] Boton presionado: Mezcla registrada. Tiempo Reset.\n");
123 }
124
125 if (estado_actual_global == ESTADO_REPOSO) {
126 color_verde();
127 if (boton_inicio_presionado()) {
128 estado_actual_global = ESTADO_MONITOREO;
129 printf("[SISTEMA] Inicio de Monitoreo...\n");
130 lcd_clear();
131 }
132 } else {
133 if (xQueueReceive(xColaDatos, &datos_vivos, 0) == pdTRUE) {
134 // Si acabamos de mezclar manualmente, anulamos la alerta visual de mezcla
135 if (mezcla_manual_hecha && datos_vivos.causa_alerta == 6) {
136 datos_vivos.causa_alerta = 0;
137 }
138
139 printf("DATA >> T:%2.1f pH:%2.2f Alerta:%d | Seg_Mezcla: %d\n",
140 datos_vivos.temp_cafe, datos_vivos.ph_valor,
141 datos_vivos.causa_alerta, seg_desde_ultima_mezcla);
142
143 if (datos_vivos.causa_alerta == 99) {
144 estado_actual_global = ESTADO_ALARMA;
145 } else if (datos_vivos.causa_alerta > 0) {
146 ticks_en_alerta++;
147 if (ticks_en_alerta >= (TIEMPO_MAX_PERSISTENCIA * 5)) estado_actual_global = ESTADO_ALARMA;
148 else estado_actual_global = ESTADO_ALERTA;
149 } else {
150 estado_actual_global = ESTADO_MONITOREO;
151 ticks_en_alerta = 0;
152 // Bajamos la bandera de mezcla tras un ciclo limpio
153 if (seg_desde_ultima_mezcla < 10) mezcla_manual_hecha = false;
154 }
155 }
156
157 toggle_visual = !toggle_visual;
158 switch(estado_actual_global) {
159 case ESTADO_MONITOREO: color_azul(); break;
160 case ESTADO_ALERTA: if (toggle_visual) color_amarillo(); else color_azul(); break;
161 case ESTADO_ALARMA: if (toggle_visual) color_rojo(); else color_azul(); break;
162 default: color_verde(); break;
163 }
164 }
165 vTaskDelay(pdMS_TO_TICKS(200));
166 }
167}
